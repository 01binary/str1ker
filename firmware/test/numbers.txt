#include <Arduino.h>
#include <Wire.h>

const uint32_t I2C_FREQUENCY_HZ = 400000;
const uint8_t LED_DRIVER_ADDRESS = 0x34;  // IS31FL3739 AD pin tied to GND.

const uint8_t REG_PWM_START = 0x03;
const uint8_t REG_PWM_END = 0x8F;
const uint8_t REG_SCALING_START = 0x92;
const uint8_t REG_SCALING_END = 0x9F;
const uint8_t REG_CONFIGURATION = 0xA0;
const uint8_t REG_GLOBAL_CURRENT = 0xA1;
const uint8_t REG_PULL_RESISTORS = 0xB0;
const uint8_t REG_PWM_FREQUENCY = 0xB2;
const uint8_t REG_RESET = 0xCF;

const uint8_t RESET_COMMAND = 0xAE;
const uint8_t CONFIG_8_SW_8_CS_NORMAL = 0x11;
const uint8_t GLOBAL_CURRENT_TEST = 0x20;
const uint8_t PULL_RESISTORS_2K_WHEN_OFF = 0x33;
const uint8_t PWM_FREQUENCY_32_KHZ = 0x01;
const uint8_t LED_ON = 0xFF;
const uint8_t LED_OFF = 0x00;

const uint8_t SEG_A = 1 << 0;
const uint8_t SEG_B = 1 << 1;
const uint8_t SEG_C = 1 << 2;
const uint8_t SEG_D = 1 << 3;
const uint8_t SEG_E = 1 << 4;
const uint8_t SEG_F = 1 << 5;
const uint8_t SEG_G = 1 << 6;
const uint8_t SEG_DP = 1 << 7;

const uint8_t DIGIT_SEGMENTS[] =
{
  SEG_A | SEG_B | SEG_C | SEG_D | SEG_E | SEG_F,         // 0
  SEG_B | SEG_C,                                         // 1
  SEG_A | SEG_B | SEG_D | SEG_E | SEG_G,                 // 2
  SEG_A | SEG_B | SEG_C | SEG_D | SEG_G,                 // 3
  SEG_B | SEG_C | SEG_F | SEG_G,                         // 4
  SEG_A | SEG_C | SEG_D | SEG_F | SEG_G,                 // 5
  SEG_A | SEG_C | SEG_D | SEG_E | SEG_F | SEG_G,         // 6
  SEG_A | SEG_B | SEG_C,                                 // 7
  SEG_A | SEG_B | SEG_C | SEG_D | SEG_E | SEG_F | SEG_G, // 8
  SEG_A | SEG_B | SEG_C | SEG_D | SEG_F | SEG_G          // 9
};

void writeRegister(uint8_t reg, uint8_t value)
{
  Wire.beginTransmission(LED_DRIVER_ADDRESS);
  Wire.write(reg);
  Wire.write(value);
  Wire.endTransmission();
}

void fillRegisters(uint8_t firstReg, uint8_t lastReg, uint8_t value)
{
  for (uint8_t reg = firstReg; reg <= lastReg; reg++)
  {
    writeRegister(reg, value);
  }
}

uint8_t pwmRegisterFor(uint8_t sw, uint8_t cs)
{
  const uint8_t csOffsets[] = {0x00, 0x01, 0x02, 0x03, 0x04, 0x0B, 0x0C, 0x0D};
  return REG_PWM_START + ((sw - 1) * 0x10) + csOffsets[cs - 1];
}

void writeDigit(uint8_t sw, uint8_t segments)
{
  for (uint8_t cs = 1; cs <= 8; cs++)
  {
    if (segments & (1 << (cs - 1)))
    {
      writeRegister(pwmRegisterFor(sw, cs), LED_ON);
    }
  }
}

void writeDisplay(uint8_t display, uint8_t leftDigit, uint8_t rightDigit)
{
  const uint8_t leftSw = (display * 2) + 1;
  writeDigit(leftSw, DIGIT_SEGMENTS[leftDigit] | SEG_DP);
  writeDigit(leftSw + 1, DIGIT_SEGMENTS[rightDigit]);
}

void setup()
{
  Wire.begin();
  Wire.setClock(I2C_FREQUENCY_HZ);

  writeRegister(REG_RESET, RESET_COMMAND);
  delay(1);

  writeRegister(REG_CONFIGURATION, CONFIG_8_SW_8_CS_NORMAL);
  writeRegister(REG_GLOBAL_CURRENT, GLOBAL_CURRENT_TEST);
  writeRegister(REG_PULL_RESISTORS, PULL_RESISTORS_2K_WHEN_OFF);
  writeRegister(REG_PWM_FREQUENCY, PWM_FREQUENCY_32_KHZ);

  fillRegisters(REG_SCALING_START, REG_SCALING_END, LED_ON);
  fillRegisters(REG_PWM_START, REG_PWM_END, LED_OFF);

  writeDisplay(0, 0, 0);
  writeDisplay(1, 0, 2);
  writeDisplay(2, 0, 5);
  writeDisplay(3, 1, 0);
}

void loop()
{
}
